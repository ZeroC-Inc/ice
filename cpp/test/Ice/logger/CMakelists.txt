if(BUILD_ICE_CPP98)
	set(LOGGER_DEPS 
		ice
		testcommon
	)
elseif(BUILD_ICE_CPP11)
	set(LOGGER_DEPS 
		ice++11
		testcommon++11
	)
endif()

add_executable(iceloggerclient1)

target_sources(iceloggerclient1
	PRIVATE
		Client1.cpp
)

target_include_directories(iceloggerclient1
	PRIVATE
		${PROJECT_SOURCE_DIR}/cpp/include
		${PROJECT_SOURCE_DIR}/cpp/src
		${PROJECT_SOURCE_DIR}/cpp/test/include
		${CMAKE_CURRENT_SOURCE_DIR}
		${GENERATED_SLICES_DIR}
)

if(BUILD_ICE_CPP98)
	set_property(TARGET iceloggerclient1 PROPERTY CXX_STANDARD 98)
elseif(BUILD_ICE_CPP11)
	set_property(TARGET iceloggerclient1 PROPERTY CXX_STANDARD 11)
endif()

target_link_libraries(iceloggerclient1 PRIVATE ${LOGGER_DEPS})

target_compile_definitions(iceloggerclient1 PRIVATE ${ICE_COMPILE_DEFS})

if(MSVC)
	target_compile_options(iceloggerclient1
		PRIVATE
			${ICE_MSVC_COMPILE_OPTIONS}
	)

	set_property(TARGET iceloggerclient1 PROPERTY LINK_FLAGS ${ICE_MSVC_DLL_LINK_OPTIONS})
endif()

install(TARGETS iceloggerclient1 DESTINATION tests/bin)
